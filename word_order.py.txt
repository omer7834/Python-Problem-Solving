# Enter your code here. Read input from STDIN. Print output to STDOUT
from collections import OrderedDict

n = int(input())
word_counts = OrderedDict()

for _ in range(n):
    word = input()
    # Use .get(key, 0) to get the current count or 0 if it's the first time
    word_counts[word] = word_counts.get(word, 0) + 1

# Print the number of unique words
print(len(word_counts))

# Print the counts of each word in the order of their first appearance
# The * operator unpacks the list of values into separate arguments for print
print(*word_counts.values())
